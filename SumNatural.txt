
import java.util.Scanner;

public class SumNatural {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("Enter number of term : ");
        int n = sc.nextInt();
        int sum = 0;
        for(int i = 0; i<=n; i++) {
            sum+=i;
        }
        System.out.println("Sum of the n is : " + sum );
        
    }

}
